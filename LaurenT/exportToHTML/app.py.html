from flask import Flask, render_template, request 
 
import re 
 
def to_words(text): 
    return re.findall(r'\w+', text) 
 
app = Flask(__name__) # creates Flask object with the __name__ attribute (which is __main__) 
 
# The following lines beginning with @ are known as a Python "decorators". 
# Each takes the function defined immediately below it and wraps it up in a new function that calls it 
# when the user visits a particular page of your application specified by the URL in quotes. 
# Using different @app.route calls, you can make the framework automatically 'trigger' different 
# parts of the code when the user visits different parts of your application or a POST is sent to that page. 
 
@app.route("/") # makes landingPage() execute when you navigate to root directory of local host 
def landingPage(): 
    return render_template( 
        'formEntry.html', message="") 
 
@app.route('/', methods=['POST']) 
 
 
def processUserInput(): 
    n1 = str(request.form['n1']) 
    n2 = str(request.form['n2']) 
 
    addA = "" 
    addB = "" 
    for i in range(len(to_words(n1))): 
        addA = addA + (to_words(n1)[i])[1:] + (to_words(n1)[i])[0] + "ay " 
    n1 = addA 
 
 
    for i in range(len(to_words(n2))): 
        addB = addB + (to_words(n2)[i])[-3] + (to_words(n2)[i])[:-3] 
    n2 = addB 
 
    total = n1 + n2 
 
 
    myMessage = " " + str(total) 
 
    return render_template( 
        'responseScreen.html',message=myMessage) 
 
 
if __name__ == "__main__": # prevents code in this file from running if imported by another Python script 
    # starts the development server for Flask and allows users to visit 
    # this web application from the local machine by visiting localhost 
    app.run()
